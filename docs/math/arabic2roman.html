arabic2roman

arabic2roman( $integer )

Description

Convert an Arabic numeral (integer) to Roman numerals

Parameters

  • $integer [int] - Integer to convert into Roman numerals

Returns

  • [string] - A string representing the integer in Roman numerals

function arabic2roman( $integer ){ 
    
    $table = array('M'=>1000, 'CM'=>900, 'D'=>500, 'CD'=>400, 'C'=>100, 'XC'=>90, 'L'=>50, 'XL'=>40, 'X'=>10, 'IX'=>9, 'V'=>5, 'IV'=>4, 'I'=>1); 
    $roman = ''; 
    
    while($integer > 0){
        foreach($table as $rom=>$arb){ 
            if($integer >= $arb){ 
                $integer -= $arb; 
                $roman .= $rom; 
                break; 
            } 
        } 
    } 

    return $roman; 
}
