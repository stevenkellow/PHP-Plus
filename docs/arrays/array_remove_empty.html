array_remove_empty

array_remove_empty( $arr, $reindex = false )

Description

Remove null or empty elements from an array

Parameters

  • $arr [array] - Array to check for empty values
  • $reindex [bool] - Optionally reindex array after empty values removed

Returns

  • $narr [array] - Array without empty values

Source code

Author: Jonas John

function array_remove_empty( $arr, $reindex = false ){
    
    $narr = array();
    while( list( $key, $val ) = each( $arr ) ){
        if( is_array( $val ) ){
            $val = array_remove_empty( $val );
            // does the result array contain anything?
            if( count( $val )!=0 ){
                // yes :-)
                $narr[$key] = $val;
            }
        }
        else {
            if( trim( $val ) != ""){
                $narr[$key] = $val;
            }
        }
    }
    unset( $arr );
    
    if( $reindex == true ){
        $narr = array_values( $narr ); // 'reindex' array
    }
    
    return $narr;
}
